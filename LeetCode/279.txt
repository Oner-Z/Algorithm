class Solution {
public:
    int numSquares(int n) {
        int res[n + 5];
        memset(res, 0, n + 5);
        // vector<int> res(n + 1);
        for (int i = 1; i <= n; i ++ ){
            int m = INT_MAX;
            for (int j = 1; j * j <= i; j ++ ){
                m = min(m, res[i - j * j]);
            }
            res[i] = m + 1;
        }
        return res[n];
    }
};
