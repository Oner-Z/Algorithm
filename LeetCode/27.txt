class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int s = nums.size();
        int l = 0, r = 0;
        while (r < s){
            if (nums[r] != val){
                nums[l] = nums[r];
                l ++ ;
            }
            r ++ ;
        }
        return l;
    }
};
